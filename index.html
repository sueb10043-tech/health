JSON.parse(recordsStr) : {}; const schoolInfo = JSON.parse(scriptProps.getProperty(SCHOOL_KEY) || 'null'); return ContentService.createTextOutput(JSON.stringify({ records, schoolInfo })) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const scriptProps = PropertiesService.getScriptProperties(); let data; try { data = JSON.parse(e.postData.contents); } catch (err) { return createJsonResponse({ result: "error", message: "JSON_PARSE_ERROR: " + err.toString() }); } if (data.type === "settings") { scriptProps.setProperty(SCHOOL_KEY, JSON.stringify(data.schoolInfo)); return createJsonResponse({ result: "success" }); } if (data.type === "analyze") { return runGeminiAnalysis(data.prompt, data.dataStr); } if (data.date) { const currentRecordsStr = scriptProps.getProperty(DATA_KEY); let records = currentRecordsStr ? JSON.parse(currentRecordsStr) : {}; records[data.date] = data; scriptProps.setProperty(DATA_KEY, JSON.stringify(records)); return createJsonResponse({ result: "success" }); } return createJsonResponse({ result: "error", message: "UNKNOWN_ACTION_TYPE" }); } function createJsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } /** * [수정] 사용 가능한 제미나이 모델을 유동적으로 찾는 함수 */ function getAvailableModel() { const listUrl = "https://generativelanguage.googleapis.com/v1beta/models?key=" + API_KEY; try { const response = UrlFetchApp.fetch(listUrl, { method: "get", muteHttpExceptions: true }); const resText = response.getContentText(); const data = JSON.parse(resText); if (data.models && data.models.length > 0) { // 1. generateContent를 지원하는 모델 필터링 const validModels = data.models.filter(m => m.supportedGenerationMethods.includes("generateContent")); // 2. 가장 안정적인 1.5 flash 모델 우선 선택, 없으면 첫 번째 가용 모델 const bestFit = validModels.find(m => m.name.includes("gemini-1.5-flash")) || validModels[0]; return bestFit.name; } } catch (e) { console.error("Model discovery failed: " + e); } return "models/gemini-1.5-flash"; // 최후의 보루 } function runGeminiAnalysis(userPrompt, historyData) { const modelName = getAvailableModel(); const url = `https://generativelanguage.googleapis.com/v1beta/${modelName}:generateContent?key=${API_KEY}`; const payload = { contents: [{ parts: [{ text: `사용자 질문: ${userPrompt}\n\n최근 기록:\n${historyData}` }] }], systemInstruction: { parts: [{ text: "당신은 건강 전문가입니다. 혈당(아침 100이하, 식후 180이하 목표) 데이터를 기반으로 친절하게 분석하세요." }] } }; try { const res = UrlFetchApp.fetch(url, { method: "post", contentType: "application/json", payload: JSON.stringify(payload), muteHttpExceptions: true }); const resCode = res.getResponseCode(); const resText = res.getContentText(); const result = JSON.parse(resText); if (resCode === 200 && result.candidates && result.candidates[0] && result.candidates[0].content) { return createJsonResponse({ answer: result.candidates[0].content.parts[0].text }); } else { // 상세 에러 정보 반환 const errorMsg = result.error ? `[${result.error.status}] ${result.error.message}` : "UNKNOWN_API_ERROR"; return createJsonResponse({ result: "error", message: `API_RESPONSE_CODE_${resCode}: ${errorMsg}`, debug: resText.substring(0, 200) }); } } catch (e) { return createJsonResponse({ result: "error", message: "SERVER_FETCH_EXCEPTION: " + e.toString() }); } }
